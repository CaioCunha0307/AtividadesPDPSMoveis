package com.example.exemploseekbar;


import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.RatingBar;
import android.widget.SeekBar;
import android.widget.TextView;
import android.widget.Toast;


import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;


public class MainActivity extends AppCompatActivity {
    TextView tvVolume, tvAvaliacao;
    SeekBar seekBarVolume;
    RatingBar ratingBarAvaliacao;
    Button btnAvaliar;




    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);
        tvVolume = (TextView) findViewById(R.id.lblVolume);
        seekBarVolume = (SeekBar) findViewById(R.id.sbVolume);
        ratingBarAvaliacao = (RatingBar) findViewById(R.id.rbAvaliacao);
        tvAvaliacao = (TextView) findViewById(R.id.lblAvaliacao);
        btnAvaliar = (Button) findViewById(R.id.cmdAvaliar);
        btnAvaliar.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                String avaliacao;
                avaliacao = String.valueOf(ratingBarAvaliacao.getRating());
                tvAvaliacao.setText("Avaliação : " + avaliacao + " estrelas");


            }
        });
        seekBarVolume.setOnSeekBarChangeListener(new SeekBar.OnSeekBarChangeListener() {
            @Override
            public void onProgressChanged(SeekBar seekBar, int progress, boolean fromUser) {
                tvVolume.setText("Volume : " +progress);
            }


            @Override
            public void onStartTrackingTouch(SeekBar seekBar) {
                Toast.makeText(getApplicationContext(),"O Seekbar foi iniciado", Toast.LENGTH_SHORT).show();


            }


            @Override
            public void onStopTrackingTouch(SeekBar seekBar) {
                Toast.makeText(getApplicationContext(),"O Seekbar foi parado", Toast.LENGTH_SHORT).show();


            }
        });


    }
}
